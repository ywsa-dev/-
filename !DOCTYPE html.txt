async function parseEntFile(data){
  let buffer=data.buffer||data;

  // 1. GZIP 해제 시도 (project ent.html 방식: 네이티브 DecompressionStream)
  try{
    const ds=new DecompressionStream('gzip');
    const decompressedStream=new Response(buffer).body.pipeThrough(ds);
    buffer=await new Response(decompressedStream).arrayBuffer();
  }catch(e){
    // GZIP이 아니면 그냥 진행
  }

  // 2. TAR 내의 project.json 및 에셋 탐색
  const view=new Uint8Array(buffer);
  let offset=0;
  let projectJson=null;
  const assets={};

  while(offset<view.length-512){
    let fileName='';
    for(let i=0;i<100;i++){
      if(view[offset+i]===0)break;
      fileName+=String.fromCharCode(view[offset+i]);
    }
    if(fileName.length===0)break;

    let sizeStr='';
    for(let i=0;i<12;i++){
      sizeStr+=String.fromCharCode(view[offset+124+i]);
    }
    const size=parseInt(sizeStr.trim(),8)||0;
    const type=String.fromCharCode(view[offset+156]);

    offset+=512;

    if(type==='0'||type===''||type==='\0'||type==='\x00'){
      const fileData=view.slice(offset,offset+size);
      const trimmedName=fileName.trim();

      if(trimmedName.endsWith('project.json')){
        const jsonText=new TextDecoder().decode(fileData);
        projectJson=JSON.parse(jsonText);
      }else if(fileData&&fileData.length>0){
        const p=trimmedName.replace(/\\/g,'/').split('/');
        assets[p.slice(-2).join('/')]=fileData;
        assets[p[p.length-1]]=fileData;
        assets[trimmedName.replace(/\\/g,'/')]=fileData;
      }
    }

    offset+=Math.ceil(size/512)*512;
  }

  if(!projectJson)throw new Error('project.json을 찾을 수 없습니다.');
  return{projectJson,assets};
}

function uint8ToBase64(d){
  let b='';const c=8192;
  for(let i=0;i<d.length;i+=c)
    b+=String.fromCharCode.apply(null,d.subarray(i,i+c));
  return btoa(b);
}

function resolveAsset(id,fileurl,filename){
  var tries=[id,fileurl,filename];
  for(var i=0;i<tries.length;i++){
    var v=tries[i]; if(!v) continue;
    if(ASSETS[v]) return ASSETS[v];
    var parts=String(v).replace(/\\\\\\\\/g,'/').split('/');
    var k2=parts.slice(-2).join('/'); if(ASSETS[k2]) return ASSETS[k2];
    var fn=parts[parts.length-1]; if(ASSETS[fn]) return ASSETS[fn];
    for(var k in ASSETS){

function isTouching(a,b){
  var aw=a.costumes[a.curCostume]?a.costumes[a.curCostume].width*a.scaleX/2:24;
  var ah=a.costumes[a.curCostume]?a.costumes[a.curCostume].height*a.scaleY/2:24;
  var bw=b.costumes[b.curCostume]?b.costumes[b.curCostume].width*b.scaleX/2:24;
  var bh=b.costumes[b.curCostume]?b.costumes[b.curCostume].height*b.scaleY/2:24;

function isOnWall(sp){

function spawn(blocks,sp,onDone){

function blocksOf(script){
  if(!script)return[];
  if(typeof script==='string'){try{script=JSON.parse(script);}catch(e){return[];}}
  // Entry wraps scripts in {script:[...]} objects
  if(script&&!Array.isArray(script)&&script.script){script=script.script;}
  if(!Array.isArray(script)||!script.length)return[];
  if(Array.isArray(script[0]))return[].concat.apply([],script);
  return script;
}
function stmts(b,idx){
  if(!b)return[];
  var s=b.statements;
  if(!s)return[];
  // statements can be object {0:[...]} or array [[...]]
  if(Array.isArray(s)){return Array.isArray(s[idx])?s[idx]:[];}
  if(s[idx]){return Array.isArray(s[idx])?s[idx]:(s[idx].script?s[idx].script:[]);}
  return[];
}
